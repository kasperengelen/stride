157 template <typename LocationType>
158 void GeoGridBase<LocationType>::Finalize()
159 {
160  std::vector<geogrid_detail::KdTree2DPoint> points;
161  for (const auto& loc : m_locations) {
162  points.emplace_back(geogrid_detail::KdTree2DPoint(loc.get()));
163  }
164  m_tree = GeoGridKdTree::Build(points);
165  m_finalized = true;
166 }
